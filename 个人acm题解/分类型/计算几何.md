```c++
//叉乘
Vector multi(Vector a,Vector b){return Vector(a.y*b.z-b.y*a.z,b.x*a.z-a.x*b.z,a.x*b.y-b.x*a.y);}
//转单位向量
Vector to_unitVector(Vector u){
    double k=sqrt(Dot(u,u));
    return Vector(u/k);
}
```
